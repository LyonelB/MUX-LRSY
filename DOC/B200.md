Installation

    $ cd
    $ mkdir workarea-uhd
    $ cd workarea-uhd
    $ git clone https://github.com/EttusResearch/uhd
    $ cd uhd
    $ git tag -l
    $ git checkout release_003_010_003_000
    $ cd host
    $ mkdir build
    $ cd build
    $ cmake ../
    $ make
    $ make test
    $ sudo make install
    $ sudo ldconfig
    $ export LD_LIBRARY_PATH=/usr/local/lib
